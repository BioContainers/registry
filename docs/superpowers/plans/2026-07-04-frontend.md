# BioContainers Frontend (Vue 3 + Vite) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A backend-free Vue 3 + Vite SPA that reproduces the three surviving biocontainers.pro screens (Home/stats, Registry search, Tool detail) reading only the static `data/` catalog, with client-side MiniSearch, deployed to GitHub Pages at biocontainers.pro.

**Architecture:** New app in `web/` (leaves the legacy Vue 2 app untouched until cut-over). A thin `dataClient` fetches the static JSON contract; a `search` module wraps MiniSearch over `search-index.json`. Three views + Nav render it. Deploy workflow builds Vite and copies repo-root `data/` into the Pages artifact.

**Tech Stack:** Vue 3, Vite 5, Vue Router 4, View UI Plus (UI kit), MiniSearch, Vitest + @vue/test-utils.

## Global Constraints

- Node 20+. Vue 3 + Vite only; no webpack, no `vue-resource`, no Vuex.
- The app reads **only** the static contract at `${import.meta.env.BASE_URL}data/`:
  `search-index.json`, `tools/<id>.json`, `facets.json`, `stats.json`. No API calls, ever.
  Contract is defined in `docs/superpowers/specs/2026-07-04-biocontainers-static-site-design.md`.
- Routes preserved: `/` (Home), `/registry` (search), `/tools/:id` (detail), `*` (NotFound).
  History mode; `404.html` + CNAME preserved for Pages deep-linking.
- Container install commands come straight from the data (`versions[].containers[]`); the app
  never constructs registry URLs itself.
- Legacy `src/`, `build/`, `config/`, webpack `package.json` stay until Task 9 (cut-over).

```js
// web/src/lib/dataClient.js
const base = () => (import.meta.env?.BASE_URL || '/')
export const dataUrl = (path) => `${base()}data/${path}`

let _index, _facets, _stats
const _tools = new Map()

async function getJson(url) {
  const r = await fetch(url)
  if (!r.ok) throw new Error(`fetch ${url} -> ${r.status}`)
  return r.json()
}

export async function loadIndex() { return (_index ??= await getJson(dataUrl('search-index.json'))) }
export async function loadFacets() { return (_facets ??= await getJson(dataUrl('facets.json'))) }
export async function loadStats() { return (_stats ??= await getJson(dataUrl('stats.json'))) }
export async function loadTool(id) {
  if (!_tools.has(id)) _tools.set(id, await getJson(dataUrl(`tools/${id}.json`)))
  return _tools.get(id)
}
export function __reset() { _index = _facets = _stats = undefined; _tools.clear() }
```

```js
// web/src/lib/search.js
import MiniSearch from 'minisearch'

const STORE = ['id','name','description','license','toolclass','registries','latest_version','versionCount','total_pulls']

export function buildIndex(records) {
  const mini = new MiniSearch({
    fields: ['name', 'description'],
    storeFields: STORE,
    searchOptions: { boost: { name: 3 }, prefix: true, fuzzy: 0.2 },
  })
  mini.addAll(records)
  return mini
}

export function runSearch(mini, query) {
  const q = (query || '').trim()
  if (!q) return null
  return mini.search(q).map((h) => {
    const doc = {}
    for (const k of STORE) doc[k] = h[k]
    return doc
  })
}
```

```js
// web/src/lib/registryState.js
export function applyFacet(records, facetName, facetValue) {
  if (!facetName || !facetValue) return records
  if (facetName === 'registry') return records.filter((r) => (r.registries || []).includes(facetValue))
  return records.filter((r) => r[facetName] === facetValue)
}

export function sortRecords(records, key, order) {
  const dir = order === 'asc' ? 1 : -1
  return [...records].sort((a, b) => {
    const av = a[key], bv = b[key]
    if (typeof av === 'string') return av.localeCompare(bv) * dir
    return (av - bv) * dir
  })
}

export function paginate(records, page, pageSize) {
  const start = (page - 1) * pageSize
  return records.slice(start, start + pageSize)
}
```

```js
// web/src/lib/toolView.js
export function installFlags(tool) {
  const types = new Set()
  for (const v of tool.versions || []) for (const c of v.containers || []) types.add(c.type)
  return { conda: types.has('conda'), docker: types.has('docker'), singularity: types.has('singularity') }
}

export function condaCommand(tool) {
  const v = (tool.versions || [])[0]
  return v?.containers?.find((c) => c.type === 'conda')?.command || null
}
```

```yaml
    needs: build
    runs-on: ubuntu-latest

      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    steps:
      - id: deployment
        uses: actions/deploy-pages@v4
```
